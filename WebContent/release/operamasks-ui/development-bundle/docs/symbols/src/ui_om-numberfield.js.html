 16     // 设置小数精度
 17     var fixPrecision = function(value, c, p) {
 18         var v = value.indexOf("."),n;       
 19         if (isNaN(value) && value != ".") {
 20             for (; isNaN(value);) {
 21                 value = value.substring(0, value.length - 1);
 22             }
 23         }
 24         if(!p.allowNegative && (n=value.indexOf("-"))!= -1){
 25         	var array=value.split("-");
 26         	value=array.join("");
 27         }
 28         if(!p.allowDecimals&&v!=-1){
 29             return value.substring(0, v);
 30          }
 31         
 32         if(v!=-1){
 33             value=value.substring(0,v+p.decimalPrecision+1);
 34         }
 35         return value;
 36         
 37     };
 38 
 39     /** 
 40      * @name omNumberField
 41      * @author 王璠
 42      * @class 数字输入框组件，只能输入数字，字符自动过滤掉。<br/>
 43      * @constructor
 44      * @description 构造函数. 
 45      * @param p 标准config对象：{}
 46      * @example
 47      * $('numberFielddiv').omNumberField({decimalPrecision:3});
 48      */
 49     $.widget("om.omNumberField", {
 50         options: /** @lends omNumberField.prototype */ 
 51         {
 52             /**
 53              * 是否允许输入小数。
 54              * @default true
 55              * @type Boolean
 56              * @example
 57              * $('#input').omNumberField({allowDecimals:true});
 58              */
 59             allowDecimals: true,  //是否允许输入小数
 60             /**
 61              * 是否允许输入负数。
 62              * @default true
 63              * @type Boolean
 64              * @example
 65              * $('#input').omNumberField({allowNegative:true});
 66              */
 67             allowNegative: true,  //是否允许输入负数
 68             /**
 69              * 精确到小数点后几位。
 70              * @default 2
 71              * @type Number
 72              * @example
 73              * $('#input').omNumberField({decimalPrecision:2});
 74              */
 75             decimalPrecision: 2, //精确到小数点后几位
 76             /**
 77              * 是否禁用组件。
 78              * @default false
 79              * @type Boolean
 80              * @example
 81              * $('#input').omNumberField({disabled:true});
 82              */
 83             disabled: false,
 84             /**
 85              * 是否只读。
 86              * @default false
 87              * @type Boolean
 88              * @example
 89              * $('#input').omNumberField({readOnly:true});
 90              */
 91             readOnly: false            
 92         },
 93 
 94         _create : function() {
 95             // 允许输入的字符
 96             var options = this.options;
 97             this.element.addClass('om-numberfield om-widget om-state-default om-state-nobg');
 98 
 99             if (typeof options.disabled !== "boolean") {
100                 this.options.disabled = this.element.prop("disabled");
101             }
102 
103             if (typeof options.readOnly !== "boolean") {
104                 this.options.readOnly = this.element.attr("readOnly");
105             }
106 
107             if (this.element.is(":disabled")) {
108                 this.options.disabled = true;
109             }
110             this.element.bind('keypress',function(e) {
111                 if (e.which == null && (e.charCode != null || e.keyCode != null)) {
112                     e.which = e.charCode != null ? e.charCode : e.keyCode;
113                 }
114                 var k = e.which;
115                 if (k === 8 || (k == 46 && e.button == -1) || k === 0) {
116                     return;
117                 }
118                 var character = String.fromCharCode(k);
119                 $.data(this,"character",character);
120                 var value = $(this).val()+character;
121                 var allowed = $.data(this, "allowed");
122                 if (allowed.indexOf(character) === -1||($(this).val().indexOf("-") !== -1 && character == "-")
123                         || ($(this).val().indexOf(".") !== -1 && character == ".")) {
124                     e.preventDefault();
125                 }
126             }).bind('focus',function(){
127                $(this).addClass('om-state-focus');
128             }).bind('blur',function(){
129                 $(this).removeClass('om-state-focus');
130                var character = $.data(this,"character");
131                this.value=fixPrecision(this.value, character, options)
132             }).bind('mouseenter mouseout',function(){      
133                 $(this).toggleClass("om-state-hover");
134             });
135             this._setOption("disabled", options.disabled);
136         },
137 
138         _setOption : function(key, value) {
139           //  $.Widget.prototype._setOption.apply(this, arguments);
140             this._buildAllowChars();
141             if (key === "disabled") {
142                 if (value) {
143                     this.element.attr("disabled", true);
144                    // this.element.removeClass("om-state-nobg");
145                     this.element.addClass("om-numberfield-disabled");
146                 } else {
147                     this.element.attr("disabled", false);
148                     this.element.removeClass("om-numberfield-disabled");
149                 }
150                 return;
151             }
152         },
153 
154         _buildAllowChars : function() {
155             var allowed = "0123456789";
156 
157             // 允许输入的字符
158             if (this.options.allowDecimals) {
159                 allowed = allowed + ".";
160             }
161             if (this.options.allowNegative) {
162                 allowed = allowed + "-";
163             }
164             if (this.options.readOnly) {
165                 allowed = "";
166             }
167             $.data(this.element[0], "allowed", allowed);
168         }
169         /**
170          * 禁用组件。
171          * @name omNumberField#disable
172          * @function
173          * @example
174          * $('#input').omNumberField("disable")
175          */
176 
177         /**
178          * 启用组件。
179          * @name omNumberField#enable
180          * @function
181          * @example
182          * $('#input').omNumberField("enable")
183          */
184     });
